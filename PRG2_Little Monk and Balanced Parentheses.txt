//Little Monk and Balanced Parentheses
//https://www.hackerearth.com/challenges/competitive/codemonk-checkpoint-i-1/algorithm/little-monk-and-balanced-parentheses/
#include <iostream>
#include <bits/stdc++.h>
using namespace std;

using namespace std;

int main() {
	int n;
	cin >> n;										// Reading input from STDIN
	int a[n+1];
	for(int i=1;i<=n;i++) cin>>a[i];
	stack<int>st;
	st.push(0);
	int len=0;
	for(int i=1;i<=n;i++)
	{
		if(a[i]>0)
		{
			st.push(i);
			continue;
		}
		if(st.top()==-1*(a[i]))
		{
			st.pop();
			len=max(len,i-st.top());
		}
		else{
			st.push(i);
		}
	}
		cout<<len<<"\n";
		return 0;
	}
